-- login table 생성
create table login(
	login_no 		number(3)  							-- 정수 0~999 저장 가능
	,login_id  		varchar2(20) 	not null unique		-- 로그인 아이디 저장
	,pwd			varchar2(20)	not null  			-- 암호 저장
	,primary key(login_no)
);
